def cum_return((n1, n2)):
    # trends
    tdf['t1'] = func(tdf['closeAsk'], int(n1))#, win_type=win_type)
    tdf['t2'] = func(tdf['closeAsk'], int(n2))#, win_type=win_type)
    # differences
    tdf['difference'] = (tdf.t1 - tdf.t2)
    # positions
    tdv[:] = 0.00
    tdf['position'] = np.where(tdf['difference'] > tdv, 1, -1)
    tdf['strategy'] = tdf.position.shift(1) * tdf.returns
    cr = -np.exp(tdf['strategy'].cumsum())[-1]
    return cr
